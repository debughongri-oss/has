    **3. 创建 `miniprogram/pages/admin/reviews/list.js`：**

    遵循管理端页面模式（isArtist 权限检查 + onShow 刷新）：

    ```javascript
    const reviewsService = require('../../../services/reviews')
    const authService = require('../../../services/auth')

    Page({
      data: {
        reviews: [],
        loading: true,
        hasMore: true,
        page: 1
      },

      onLoad: function () {
        if (!authService.isArtist()) {
          wx.showToast({ title: '无权限访问', icon: 'none' })
          setTimeout(() => wx.navigateBack(), 1500)
          return
        }
        this.loadReviews()
      },

      onPullDownRefresh: function () {
        this.setData({ page: 1 })
        this.loadReviews(true).then(() => wx.stopPullDownRefresh())
      },

      onReachBottom: function () {
        if (this.data.hasMore) {
          this.setData({ page: this.data.page + 1 })
          this.loadReviews()
        }
      },

      loadReviews: function (reset = false) {
        const page = reset ? 1 : this.data.page
        this.setData({ loading: reset })
        return reviewsService.getReviewsList(page, 10)
          .then(result => {
            // 格式化时间显示
            const reviews = result.list.map(r => ({
              ...r,
              timeLabel: this.formatTime(r.created_at)
            }))
            this.setData({
              reviews: reset ? reviews : this.data.reviews.concat(reviews),
              hasMore: result.hasMore,
              loading: false
            })
          })
          .catch(err => {
            console.error('加载评价列表失败:', err)
            this.setData({ loading: false })
          })
      },

      formatTime: function (date) {
        if (!date) return ''
        const d = new Date(date)
        return `${d.getFullYear()}-${String(d.getMonth() + 1).padStart(2, '0')}-${String(d.getDate()).padStart(2, '0')}`
      }
    })
    ```
