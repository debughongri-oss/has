    const { callCloudFunction } = require('./api')

    const createReview = async (bookingId, rating, content, userInfo) => {
      const result = await callCloudFunction('reviews', {
        action: 'create',
        booking_id: bookingId,
        rating,
        content: content || '',

    const getByBooking = async (bookingId) => {
      const result = await callCloudFunction('reviews', { action: 'getByBooking', booking_id: bookingId })
      if (result.errCode !== 0) throw new Error(result.errMsg)
      return result.data
    }

    const getReviewsList = async (page, pageSize) => {
      const result = await callCloudFunction('reviews', { action: 'list', page, pageSize })

      return result.data
    }

    module.exports = {

    ```javascript
    loadBookings: function (reset = false) {
      const page = reset ? 1 : this.data.page
      this.setData({ loading: reset })
      return bookingsService.getMyBookings(page, 10)
        .then(result => {
          const bookings = result.list.map(b => ({
            ...b,
            statusLabel: bookingsService.getStatusLabel(b.status),
            statusColor: bookingsService.getStatusColor(b.status)
          }))

          // 批量检查已完成预约的评价状态 per D-12
          const completedIds = bookings
            .filter(b => b.status === 'completed')
            .map(b => b._id)

          if (completedIds.length > 0) {
            Promise.all(
              completedIds.map(id =>
                reviewsService.getByBooking(id)
                  .then(review => ({ id, reviewed: !!review }))
                  .catch(() => ({ id, reviewed: false }))
              )
            ).then(results => {
              const reviewMap = {}
              results.forEach(r => { reviewMap[r.id] = r.reviewed })
              this.setData({ reviewMap })
            })
          }

          this.setData({
            bookings: reset ? bookings : this.data.bookings.concat(bookings),
            hasMore: result.hasMore,
            loading: false
          })
        })
        .catch(err => {
          console.error('加载预约记录失败:', err)
          this.setData({ loading: false })
        })
    }
    ```
