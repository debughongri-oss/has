4. Modify `loadDetail` — after setting booking data, load customer note. Replace the existing `.then(data => { ... })` block. Keep all existing normalization logic. After `this.setData({ booking: data, loading: false, ... })`, add customer note loading:

```javascript
loadDetail: function (id) {
  this.setData({ loading: true })
  bookingsService.getBookingDetail(id)
    .then(data => {
      // [existing normalization code stays exactly as-is]
      if (data.skin_type) {
        const SKIN_LABELS = { dry: '干性', oily: '油性', combination: '混合性', sensitive: '敏感性', unknown: '不确定' }
        data.skin_type_label = SKIN_LABELS[data.skin_type] || data.skin_type
      }
      data.dateLabel = formatDateLabel(data.booking_date)
      data = normalizeBookingMeta(data)
      this.setData({
        booking: data,
        loading: false,
        statusLabel: bookingsService.getStatusLabel(data.status),
        statusColor: bookingsService.getStatusColor(data.status),
        artistNotes: data.artist_notes || '',
        customerNote: null,
        customerTag: ''
      })

      // CUST-04: Load customer note for this booking's user_openid
      if (data.user_openid) {
        customersService.getCustomerNote(data.user_openid)
          .then(note => {
            // Compute tag from completed bookings count in the note data if available
            // (note doesn't contain completed_count; we compute tag client-side from note existence)
            // Tag will be computed from the booking data's user interaction — use detail action
            // For simplicity, tag is derived from customer detail or shown generically
            this.setData({ customerNote: note })
          })
          .catch(err => {
            console.error('加载客户备注失败:', err)
          })

        // Also load customer tag (best-effort via detail action)
        customersService.getCustomerDetail(data.user_openid)
          .then(detail => {
            this.setData({ customerTag: detail.tag || '' })
          })
          .catch(() => {/* tag is best-effort */})
      }
    })
    .catch(err => {
      console.error('加载详情失败:', err)
      this.setData({ loading: false })
    })
},
```
